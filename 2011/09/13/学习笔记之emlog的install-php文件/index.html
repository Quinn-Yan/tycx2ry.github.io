/&#42&#42 


 &#42 生成一个随机的字符串 


 &#42 


 &#42 @param int $length 


 &#42 @param boolean $special_chars 


 &#42 @return string 


 &#42/ 


function getRandStr($length = 12, $special_chars = true){ 


        $chars = ’abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789’; 


        if ( $special_chars ){ 


                $chars .= ’!@#$%^&&#42()’; 


        } 


        $randStr = ’’; 


        for ( $i = 0; $i < $length; $i++ ){ 


                $randStr .= substr($chars, mt_rand(0, strlen($chars) - 1), 1); 


        } 


        return $randStr; 


}
